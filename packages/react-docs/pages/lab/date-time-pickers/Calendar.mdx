# Calendar

## Import

```js
import {
  Calendar,
} from '@tonic-ui/react-lab';
```

## Usage

```jsx noInline

  <Box mb="4x" {...props} />
);

const firstDayOfWeekOptions = [
  0, // Sunday
  1, // Monday
  2, // Tuesday

];

  const [value, setValue] = React.useState(defaultValue);
  const changeBy = (value) => () => setValue(value);
  return [value, changeBy];
};

  'none': null,
  'invalid': new Date(''),
}[dateSelection]);

  const [dateSelection, changeDateSelectionBy] = useSelection('today');
  const [date, setDate] = React.useState(mapDateSelectionToDateObject(dateSelection));
  const [error, setError] = React.useState(null);
  const [firstDayOfWeek, changeWeekStartDayBy] = useSelection(0);

  React.useEffect(() => {
    setDate(mapDateSelectionToDateObject(dateSelection));
  }, [dateSelection]);

  return (
    <>
      <Flex mb="4x" columnGap="2x">
        <Text>
          Selected date:
        </Text>
        {error && (
          <Text color="red:50">{error}</Text>
        )}
        {!error && (
          <Text>{dateFns.isValid(date) && dateFns.format(date, 'yyyy-MM-dd')}</Text>
        )}
      </Flex>
      <Calendar
        date={date}

        onChange={(nextDate) => {

            setError(error);
          } else {
```

## Props

### Calendar

| Name | Type | Default | Description |
| :--- | :--- | :------ | :---------- |
| date | Date | | The selected date. |
| defaultDate | Date | | The default selected date. |
| firstDayOfWeek | number | 0 | The first day of the week.<br/>0 = Sunday<br/>1 = Monday<br/>2 = Tuesday<br/>3 = Wednesday<br/>4 = Thursday<br/>5 = Friday<br/>6 = Saturday |
| onChange | (date) => void | | Callback fired when the selected date changes. |
| onError | (error, date) => void | | Error-first callback fired when the date validation returns an error (or the date is valid after error). |
